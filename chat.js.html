// Importar la librería de Google AI (asegúrate de instalarla con: npm install @google/generative-ai) const { GoogleGenerativeAI } = require("@google/generative-ai"); // --- ¡IMPORTANTE! Configuración de la Clave API --- // Vercel tomará esta clave de las Variables de Entorno que configures en su dashboard. // NUNCA pongas tu clave API directamente aquí en el código. const apiKey = process.env.GEMINI_API_KEY; // Asegúrate que el nombre coincida con el que pongas en Vercel // Validar que la clave API esté configurada (importante para producción) if (!apiKey) { console.error("Error Fatal: La variable de entorno GEMINI_API_KEY no está configurada."); // En un entorno real, no deberías enviar este detalle al cliente // throw new Error("Configuración de API Key faltante en el servidor."); } // Inicializar el cliente de Google AI (solo si la clave existe) // Se hace fuera del handler para reutilizar la instancia si es posible const genAI = apiKey ? new GoogleGenerativeAI(apiKey) : null; const model = genAI ? genAI.getGenerativeModel({ model: "gemini-1.5-flash"}) : null; // Usa el modelo que prefieras // --- Handler de la Función Serverless --- // Vercel ejecutará esta función por cada petición a /api/chat export default async function handler(req, res) { // Verificar que la API Key y el modelo se inicializaron correctamente al inicio if (!apiKey || !model) { console.error("Error: Clave API o modelo de IA no inicializado."); // No dar detalles específicos del error al cliente return res.status(500).json({ error: "Error interno del servidor." }); } // 1. Permitir solo peticiones POST if (req.method !== 'POST') { console.log(`Método no permitido: ${req.method}`); // Establecer cabecera 'Allow' es buena práctica para 405 res.setHeader('Allow', ['POST']); return res.status(405).json({ error: `Método ${req.method} no permitido` }); } try { // 2. Obtener el mensaje y el tono del cuerpo de la petición // El frontend (React) enviará un JSON como: { "message": "...", "tone": "..." } const { message, tone } = req.body; // 3. Validar que el mensaje exista y no esté vacío if (!message || typeof message !== 'string' || message.trim() === '') { console.log("Petición inválida: falta el mensaje o está vacío."); return res.status(400).json({ error: 'Falta el mensaje del usuario o está vacío.' }); } // 4. (Opcional) Validar el tono si es necesario const validTones = ['amigable', 'profesional', 'serio', 'inspirador', 'resumido', 'explicativo']; const chatTone = tone && validTones.includes(tone) ? tone : 'amigable'; // Usar tono por defecto si no es válido // 5. Preparar el prompt para la IA, incluyendo el tono // Puedes ajustar este prompt para darle más contexto o instrucciones al asistente // Es buena idea darle un rol y contexto claro. const prompt = `Eres un asistente virtual llamado Epicentrum Assistant, especializado en ayudar a inmigrantes recién llegados a EE.UU. con información sobre recursos, trámites y derechos. Responde al siguiente mensaje del usuario de forma clara, concisa y útil, usando un tono ${chatTone}. Mensaje del usuario: "${message}"`; console.log(`Generando respuesta para: "${message}" con tono: ${chatTone}`); // 6. Llamar a la API de Gemini (o la IA que uses) const result = await model.generateContent(prompt); const response = await result.response; const aiText = response.text(); console.log("Respuesta de IA generada."); // 7. Enviar la respuesta de la IA de vuelta al frontend en formato JSON // El frontend esperará un objeto con una propiedad "response" res.status(200).json({ response: aiText }); } catch (error) { // 8. Manejo de errores (muy importante en producción) console.error("Error llamando a la API de IA o procesando la solicitud:", error); // No enviar detalles específicos del error al cliente por seguridad res.status(500).json({ error: 'Hubo un error al contactar al asistente. Intenta de nuevo.' }); } }
